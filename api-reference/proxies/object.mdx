### Attributes

<ResponseField name="id" type="string">
  This is the internal ID for this proxy.
</ResponseField>

<ResponseField name="object" type="string">
  String representing the objectâ€™s type. Objects of the same type share the same value.
</ResponseField>

<ResponseField name="organization_id" type="string">
  The ID of the organization that the proxy belongs to.
</ResponseField>

<ResponseField name="project_id" type="string">
  The ID of the project that the proxy belongs to.
</ResponseField>

<ResponseField name="name" type="string">
  The name of the proxy.
</ResponseField>

<ResponseField name="log_severity" type="string">
  The log severity of the proxy.
</ResponseField>

<ResponseField name="assembly_cookie_name" type="string">
  The name of the cookie used to store the Assembly session token.
</ResponseField>

<ResponseField name="assembly_aes_key" type="string">
  The AES key used to encrypt the Assembly session token.
</ResponseField>

<ResponseField name="assembly_aes_iv" type="string">
  The AES IV used to encrypt the Assembly session token.
</ResponseField>

<ResponseField name="frontend_host" type="string">
  The host of the proxy.
</ResponseField>

<ResponseField name="frontend_tls_cert" type="string">
  The TLS certificate of the proxy.
</ResponseField>

<ResponseField name="frontend_tls_key" type="string">
  The TLS key of the proxy.
</ResponseField>

<ResponseField name="frontend_force_https" type="boolean">
  Whether the proxy should force HTTPS.
</ResponseField>

<ResponseField name="frontend_hsts_max_age" type="integer">
  The HSTS max age of the proxy.
</ResponseField>

<ResponseField name="config_version" type="integer">
  The configuration version of the proxy.
</ResponseField>

<ResponseField name="configuration" type="object">
  The configuration of the proxy.
</ResponseField>

<ResponseField name="created_at" type="string">
  The creation date of the proxy.
</ResponseField>

<ResponseField name="updated_at" type="string">
  The last update date of the proxy.
</ResponseField>
