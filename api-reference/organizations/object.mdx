### Attributes

<ResponseField name="id" type="string">
  This is the internal ID for this Organization.
</ResponseField>

<ResponseField name="object" type="string">
  String representing the objectâ€™s type. Objects of the same type share the same value.
</ResponseField>

<ResponseField name="name" type="string">
  The name of the organization.
</ResponseField>

<ResponseField name="slug" type="string">
  The slug of the organization.
</ResponseField>

<ResponseField name="type" type="string">
  The type of the organization.
  
  It can be one of the following values:
  - `perso`: Personal organization
  - `pro`: Professional organization
</ResponseField>

<ResponseField name="current_billing_plan" type="string">
  The current billing plan of the organization.
</ResponseField>

<ResponseField name="card_brand" type="string">
  The brand of the card used to pay the organization.
</ResponseField>

<ResponseField name="card_last_four" type="string">
  The last four digits of the card used to pay the organization.
</ResponseField>

<ResponseField name="card_country" type="string">
  The country of the card used to pay the organization.
</ResponseField>

<ResponseField name="billing_address" type="string">
  The billing address of the organization.
</ResponseField>

<ResponseField name="billing_address_line_2" type="string">
  The billing address line 2 of the organization.
</ResponseField>

<ResponseField name="billing_city" type="string">
  The billing city of the organization.
</ResponseField>

<ResponseField name="billing_state" type="string">
  The billing state of the organization.
</ResponseField>

<ResponseField name="billing_zip" type="string">
  The billing zip of the organization.
</ResponseField>

<ResponseField name="billing_country" type="string">
  The billing country of the organization.
</ResponseField>

<ResponseField name="vat_id" type="string">
  The VAT ID of the organization.
</ResponseField>

<ResponseField name="extra_billing_information" type="string">
  The extra billing information of the organization.
</ResponseField>

<ResponseField name="trial_ends_at" type="string">
  In case of a trial, the date at which the trial ends.
</ResponseField>

<ResponseField name="created_at" type="string">
  The creation date of the organization.
</ResponseField>

<ResponseField name="updated_at" type="string">
  The last update date of the organization.
</ResponseField>
